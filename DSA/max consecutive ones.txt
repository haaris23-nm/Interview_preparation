class Solution {
    public int longestOnes(int[] nums, int k) {
        int s=0;
        int z=0;
        int m=0;
        for(int e=0;e<nums.length;e++){
            if(nums[e]==0){
                z++;
            }
            while(z>k){
                if(nums[s]==0){
                    z--;
                }
                s++;
            }
            m=Math.max(m,e-s+1);
        }
        return m;
   }
}
