class Solution {
    public int longestValidParentheses(String s) {
        int op=0;
        int cl=0;
        int m=0;
        for(int i=0;i<s.length();i++){
            if(s.charAt(i)=='('){
                op++;
            }
            else{
                cl++;
            }
            if(op==cl){
                m=Math.max(m,op*2);
            }
            else if(cl>op){
                op=0;
                cl=0;
            }
        }
        op=0;
        cl=0;
        for(int i=s.length()-1;i>=0;i--){
             if(s.charAt(i)=='('){
                op++;
            }
            else{
                cl++;
            }
            if(op==cl){
                m=Math.max(m,op*2);
            }
            else if(op>cl){
                op=0;
                cl=0;
            }
        }
        return m;
    }
}
