class Solution {
    public List<String> generateParenthesis(int n) {
        List<String> res=new ArrayList<>();
        solve("", 0, 0, n, res);
        return res;
    }
    private void solve(String str,int op,int cl,int n,List<String> res){
        if(str.length()==n*2){
            res.add(str);
            return;
        }
        if(op<n){
            solve(str+"(", op+1, cl, n, res);
        }
        if(cl<op){
            solve(str+")", op, cl+1, n, res);
        }
    }
}
